   179тЖТ   179тЖТfunction parseStringList(val: unknown): string[] {
   180тЖТ   180тЖТ  if (!val) return []
   181тЖТ   181тЖТ  // Already an array тАФ handle strings AND objects
   182тЖТ   182тЖТ  if (Array.isArray(val)) {
   183тЖТ   183тЖТ    return val
   184тЖТ   184тЖТ      .map((item) => {
   185тЖТ   185тЖТ        if (typeof item === 'string') return item.replace(/^\d+[\.\)\-]\s*/, '').trim()
   186тЖТ   186тЖТ        if (typeof item === 'object' && item !== null) {
   187тЖТ   187тЖТ          const obj = item as Record<string, unknown>
   188тЖТ   188тЖТ          return String(obj.title || obj.text || obj.description || obj.point || obj.content || '').trim()
   189тЖТ   189тЖТ        }
   190тЖТ   190тЖТ        return String(item).trim()
   191тЖТ   191тЖТ      })
   192тЖТ   192тЖТ      .filter((s) => s.length > 0 && !s.startsWith('[object'))
   193тЖТ   193тЖТ  }
   194тЖТ   194тЖТ  if (typeof val === 'string') {
   195тЖТ   195тЖТ    try {
   196тЖТ   196тЖТ      const parsed = JSON.parse(val)
   197тЖТ   197тЖТ      if (Array.isArray(parsed)) return parseStringList(parsed)
   198тЖТ   198тЖТ    } catch {
   199тЖТ   199тЖТ      // split by newlines or commas
   200тЖТ   200тЖТ      return val.split(/[\n]/).map(s => s.replace(/^\d+[\.\)\-]\s*/, '').trim()).filter(Boolean)
   201тЖТ   201тЖТ    }
   202тЖТ   202тЖТ  }
   203тЖТ   203тЖТ  return []
   204тЖТ   204тЖТ}
