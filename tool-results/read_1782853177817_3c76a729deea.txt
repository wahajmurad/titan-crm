   133→
   134→  // Aggregate stats

   410→
   411→                  {/* Row 3: Footer */}
   412→                  <div className="flex items-center justify-between mt-2.5 text-[11px] text-gray-300">

   414→                    <span>Daily limit: {campaign.dailyLimit}</span>
   415→                  </div>
   416→                </CardContent>
   417→              </Card>
   418→            )
   419→          })}
   420→        </div>
   421→      )}
   422→
   423→      {/* Edit Campaign Dialog */}
   424→      {selectedCampaign && (
   425→        <EditCampaignDialog
   426→          open={editOpen}
   427→          onOpenChange={setEditOpen}

   444→      )}
   445→    </div>

   451→function CreateCampaignDialog({ open, onOpenChange, onCreated }: {
   452→  open: boolean; onOpenChange: (o: boolean) => void; onCreated: () => void
   453→}) {
   454→  const [form, setForm] = useState<CampaignForm>(EMPTY_FORM)
   455→  const [loading, setLoading] = useState(false)
   456→
   457→  const update = (key: keyof CampaignForm, value: string) =>
   458→    setForm(f => ({ ...f, [key]: value }))
   459→
   460→  const handleCreate = async () => {
   461→    if (!form.name.trim()) {
   462→      toast.error('Campaign name is required')
   463→      return
   464→    }
   465→    setLoading(true)
   466→    try {
   467→      const res = await fetch('/api/campaigns', {
   468→        method: 'POST',
   469→        headers: { 'Content-Type': 'application/json' },
   470→        credentials: 'same-origin',
   471→        body: JSON.stringify({
   472→          ...form,
   473→          dailyLimit: parseInt(form.dailyLimit) || 20,
   474→        }),
   475→      })
   476→      if (res.ok) {
   477→        toast.success('Campaign created successfully')
   478→        setForm(EMPTY_FORM)
   479→        onOpenChange(false)
   480→        onCreated()
   481→      } else {
   482→        const d = await res.json().catch(() => ({}))
   483→        toast.error(d.error || 'Failed to create campaign')
   484→      }
   485→    } catch {
   486→      toast.error('Failed to create campaign')
   487→    }
   488→    setLoading(false)
   489→  }
   490→
   491→  return (
