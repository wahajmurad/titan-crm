     1→'use client'
     2→
     3→import { useEffect, useState, useCallback } from 'react'
     4→import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'
     5→import { Button } from '@/components/ui/button'
     6→import { Input } from '@/components/ui/input'
     7→import { Label } from '@/components/ui/label'
     8→import { Separator } from '@/components/ui/separator'
     9→import { Badge } from '@/components/ui/badge'
    10→import { Switch } from '@/components/ui/switch'
    11→import { Textarea } from '@/components/ui/textarea'
    12→import {
    13→  Select,
    14→  SelectContent,
    15→  SelectItem,
    16→  SelectTrigger,
    17→  SelectValue,
    18→} from '@/components/ui/select'
    19→import {
    20→  Dialog,
    21→  DialogContent,
    22→  DialogDescription,
    23→  DialogFooter,
    24→  DialogHeader,
    25→  DialogTitle,
    26→} from '@/components/ui/dialog'
    27→import { Save, CheckCircle, Plus, Trash2, Globe, Key, Settings, Shield, ExternalLink } from 'lucide-react'
    28→
    29→// ── Lead Provider Types ──────────────────────────────────────────────
    30→
    31→interface LeadProvider {
    32→  id: string
    33→  name: string
    34→  type: 'SERPER' | 'CUSTOM'
    35→  baseUrl: string
    36→  authMethod: 'bearer' | 'api-key' | 'header' | 'none'
    37→  apiKey: string
    38→  authHeader?: string
    39→  headers: Record<string, string>
    40→  requestParams: Record<string, string>
    41→  isActive: boolean
    42→}
    43→
    44→const SERPER_DEFAULT: Omit<LeadProvider, 'id'> = {
    45→  name: 'Serper (Google Search)',
    46→  type: 'SERPER',
    47→  baseUrl: 'https://google.serper.dev',
    48→  authMethod: 'api-key',
    49→  apiKey: '',
    50→  authHeader: 'X-API-KEY',
    51→  headers: {},
    52→  requestParams: {},
    53→  isActive: true,
    54→}
    55→
    56→const AUTH_METHOD_LABELS: Record<LeadProvider['authMethod'], string> = {
    57→  bearer: 'Bearer Token',
    58→  'api-key': 'API Key in Header',
    59→  header: 'Custom Header',
    60→  none: 'None',
    61→}
    62→
    63→const TYPE_LABELS: Record<LeadProvider['type'], string> = {
    64→  SERPER: 'Serper',
    65→  CUSTOM: 'Custom',
    66→}
    67→
    68→function generateId() {
    69→  return crypto.randomUUID()
    70→}
    71→
    72→// ── Component ────────────────────────────────────────────────────────
    73→
    74→export function SettingsView() {
    75→  // ── Existing settings state ──
    76→  const [settings, setSettings] = useState<Record<string, string>>({})
    77→  const [name, setName] = useState('')
    78→  const [currentPass, setCurrentPass] = useState('')
    79→  const [newPass, setNewPass] = useState('')
    80→  const [loading, setLoading] = useState(false)
    81→  const [saved, setSaved] = useState(false)
    82→
    83→  // ── Lead provider state ──
    84→  const [providers, setProviders] = useState<LeadProvider[]>([])
    85→  const [providersLoaded, setProvidersLoaded] = useState(false)
    86→  const [dialogOpen, setDialogOpen] = useState(false)
    87→  const [editingProvider, setEditingProvider] = useState<LeadProvider | null>(null)
    88→  const [savingProviders, setSavingProviders] = useState(false)
    89→
    90→  // Form state
    91→  const [formName, setFormName] = useState('')
    92→  const [formType, setFormType] = useState<LeadProvider['type']>('SERPER')
    93→  const [formBaseUrl, setFormBaseUrl] = useState('')
    94→  const [formAuthMethod, setFormAuthMethod] = useState<LeadProvider['authMethod']>('api-key')
    95→  const [formApiKey, setFormApiKey] = useState('')
    96→  const [formAuthHeader, setFormAuthHeader] = useState('')
    97→  const [formHeaders, setFormHeaders] = useState('')
    98→  const [formParams, setFormParams] = useState('')
    99→  const [formActive, setFormActive] = useState(true)
   100→
   101→  // ── Load all settings (existing + providers) ──
   102→  useEffect(() => {
   103→    Promise.all([fetch('/api/settings'), fetch('/api/auth')])
   104→      .then(([sRes, uRes]) => Promise.all([sRes.json(), uRes.json()]))
   105→      .then(([sData, uData]) => {
   106→        if (uData.user) setName(uData.user.name)
   107→        setSettings(sData.settings || {})
   108→
   109→        // Parse lead providers from settings
   110→        const raw = sData.settings?.lead_providers
   111→        if (raw) {
   112→          try {
   113→            setProviders(JSON.parse(raw))
   114→          } catch {
   115→            setProviders([])
   116→          }
   117→        }
   118→        setProvidersLoaded(true)
   119→      })
   120→      .catch(() => {
   121→        setProvidersLoaded(true)
   122→      })
   123→  }, [])
   124→
   125→  // ── Existing helpers ──
   126→  const updateSetting = (key: string, value: string) => {
   127→    setSettings(s => ({ ...s, [key]: value }))
   128→  }
   129→
   130→  const saveSettings = async () => {
   131→    setLoading(true)
   132→    try {
   133→      await fetch('/api/settings', {
   134→        method: 'POST',
   135→        headers: { 'Content-Type': 'application/json' },
   136→        body: JSON.stringify({ settings }),
   137→      })
   138→      setSaved(true)
   139→      setTimeout(() => setSaved(false), 2000)
   140→    } catch {}
   141→    setLoading(false)
   142→  }
   143→
   144→  const updateProfile = async () => {
   145→    if (!name) return
   146→    setLoading(true)
   147→    try {
   148→      const payload: Record<string, string> = { name }
   149→      if (currentPass && newPass) { payload.currentPassword = currentPass; payload.newPassword = newPass }
   150→      await fetch('/api/settings', { method: 'PATCH', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) })
   151→      setSaved(true)
   152→      setCurrentPass(''); setNewPass('')
   153→      setTimeout(() => setSaved(false), 2000)
   154→    } catch {}
   155→    setLoading(false)
   156→  }
   157→
   158→  // ── Lead Provider helpers ──
   159→
   160→  const resetForm = () => {
   161→    setFormName('')
   162→    setFormType('SERPER')
   163→    setFormBaseUrl('')
   164→    setFormAuthMethod('api-key')
   165→    setFormApiKey('')
   166→    setFormAuthHeader('')
   167→    setFormHeaders('')
   168→    setFormParams('')
   169→    setFormActive(true)
   170→    setEditingProvider(null)
   171→  }
   172→
   173→  const openAddDialog = () => {
   174→    resetForm()
   175→    setDialogOpen(true)
   176→  }
   177→
   178→  const openEditDialog = (provider: LeadProvider) => {
   179→    setEditingProvider(provider)
   180→    setFormName(provider.name)
   181→    setFormType(provider.type)
   182→    setFormBaseUrl(provider.baseUrl)
   183→    setFormAuthMethod(provider.authMethod)
   184→    setFormApiKey(provider.apiKey)
   185→    setFormAuthHeader(provider.authHeader || '')
   186→    setFormHeaders(JSON.stringify(provider.headers || {}, null, 2))
   187→    setFormParams(JSON.stringify(provider.requestParams || {}, null, 2))
   188→    setFormActive(provider.isActive)
   189→    setDialogOpen(true)
   190→  }
   191→
   192→  const handleQuickAddSerper = () => {
   193→    const newProvider: LeadProvider = {
   194→      ...SERPER_DEFAULT,
   195→      id: generateId(),
   196→    }
   197→    setEditingProvider(newProvider)
   198→    setFormName(newProvider.name)
   199→    setFormType(newProvider.type)
   200→    setFormBaseUrl(newProvider.baseUrl)
   201→    setFormAuthMethod(newProvider.authMethod)
   202→    setFormApiKey('')
   203→    setFormAuthHeader(newProvider.authHeader || '')
   204→    setFormHeaders('')
   205→    setFormParams('')
   206→    setFormActive(newProvider.isActive)
   207→    setDialogOpen(true)
   208→  }
   209→
   210→  const saveProviders = async (updatedProviders: LeadProvider[]) => {
   211→    setSavingProviders(true)
   212→    try {
   213→      await fetch('/api/settings', {
   214→        method: 'POST',
   215→        headers: { 'Content-Type': 'application/json' },
   216→        body: JSON.stringify({
   217→          settings: { lead_providers: JSON.stringify(updatedProviders) },
   218→        }),
   219→      })
   220→      setProviders(updatedProviders)
   221→    } catch {}
   222→    setSavingProviders(false)
   223→  }
   224→
   225→  const handleSaveProvider = async () => {
   226→    let parsedHeaders: Record<string, string> = {}
   227→    let parsedParams: Record<string, string> = {}
   228→
   229→    if (formHeaders.trim()) {
   230→      try {
   231→        parsedHeaders = JSON.parse(formHeaders)
   232→      } catch {
   233→        return
   234→      }
   235→    }
   236→
   237→    if (formParams.trim()) {
   238→      try {
   239→        parsedParams = JSON.parse(formParams)
   240→      } catch {
   241→        return
   242→      }
   243→    }
   244→
   245→    const providerData: LeadProvider = {
   246→      id: editingProvider?.id || generateId(),
   247→      name: formName.trim(),
   248→      type: formType,
   249→      baseUrl: formBaseUrl.trim(),
   250→      authMethod: formAuthMethod,
   251→      apiKey: formApiKey,
   252→      authHeader: formAuthHeader.trim() || undefined,
   253→      headers: parsedHeaders,
   254→      requestParams: parsedParams,
   255→      isActive: formActive,
   256→    }
   257→
   258→    let updatedProviders: LeadProvider[]
   259→    if (editingProvider) {
   260→      updatedProviders = providers.map(p => (p.id === editingProvider.id ? providerData : p))
   261→    } else {
   262→      updatedProviders = [...providers, providerData]
   263→    }
   264→
   265→    await saveProviders(updatedProviders)
   266→    setDialogOpen(false)
   267→    resetForm()
   268→  }
   269→
   270→  const handleDeleteProvider = async (id: string) => {
   271→    const updatedProviders = providers.filter(p => p.id !== id)
   272→    await saveProviders(updatedProviders)
   273→  }
   274→
   275→  const handleToggleActive = async (id: string) => {
   276→    const updatedProviders = providers.map(p =>
   277→      p.id === id ? { ...p, isActive: !p.isActive } : p
   278→    )
   279→    await saveProviders(updatedProviders)
   280→  }
   281→
   282→  // ── Render ──
   283→
   284→  return (
   285→    <div className="space-y-6 max-w-2xl">
   286→      <div>
   287→        <h1 className="text-xl font-semibold text-gray-900">Settings</h1>
   288→        <p className="text-sm text-gray-400 mt-0.5">Manage your profile and system configuration</p>
   289→      </div>
   290→
   291→      {/* ── Lead Providers ── */}
   292→      <Card className="border-slate-200">
   293→        <CardHeader className="pb-3">
   294→          <div className="flex items-center justify-between">
   295→            <div className="flex items-center gap-2">
   296→              <Globe className="w-4 h-4 text-blue-600" />
   297→              <CardTitle className="text-sm font-semibold">Lead Providers</CardTitle>
   298→            </div>
   299→            <Button
   300→              size="sm"
   301→              className="bg-blue-600 hover:bg-blue-700 text-white"
   302→              onClick={openAddDialog}
   303→            >
   304→              <Plus className="w-3.5 h-3.5 mr-1" />
   305→              Add Provider
   306→            </Button>
   307→          </div>
   308→          <p className="text-xs text-gray-500">
   309→            Configure data sources for lead discovery and enrichment
   310→          </p>
   311→        </CardHeader>
   312→        <CardContent>
   313→          {!providersLoaded ? (
   314→            <div className="flex items-center justify-center py-8 text-sm text-gray-400">
   315→              <Settings className="w-4 h-4 mr-2 animate-spin" />
   316→              Loading providers...
   317→            </div>
   318→          ) : providers.length === 0 ? (
   319→            <div className="text-center py-8">
   320→              <div className="inline-flex items-center justify-center w-10 h-10 rounded-full bg-blue-50 mb-3">
   321→                <Globe className="w-5 h-5 text-blue-600" />
   322→              </div>
   323→              <p className="text-sm font-medium text-gray-700">No lead providers configured</p>
   324→              <p className="text-xs text-gray-400 mt-1 mb-4">
   325→                Add a data source to start discovering leads. We recommend starting with Serper.dev for Google Search results.
   326→              </p>
   327→              <Button
   328→                size="sm"
   329→                variant="outline"
   330→                className="border-blue-200 text-blue-600 hover:bg-blue-50 hover:text-blue-700"
   331→                onClick={handleQuickAddSerper}
   332→              >
   333→                <Plus className="w-3.5 h-3.5 mr-1" />
   334→                Quick Add Serper.dev
   335→              </Button>
   336→            </div>
   337→          ) : (
   338→            <div className="space-y-3">
   339→              {providers.map(provider => (
   340→                <div
   341→                  key={provider.id}
   342→                  className="group flex items-start justify-between p-3 rounded-lg border border-slate-100 hover:border-slate-200 bg-slate-50/50 transition-colors"
   343→                >
   344→                  <div className="flex-1 min-w-0">
   345→                    <div className="flex items-center gap-2 mb-1">
   346→                      <button
   347→                        onClick={() => openEditDialog(provider)}
   348→                        className="text-sm font-medium text-gray-900 hover:text-blue-600 transition-colors truncate"
   349→                      >
   350→                        {provider.name}
   351→                      </button>
   352→                      <Badge
   353→                        variant={provider.isActive ? 'default' : 'secondary'}
   354→                        className={
   355→                          provider.isActive
   356→                            ? 'bg-blue-600 text-white text-[10px] px-1.5 py-0'
   357→                            : 'text-[10px] px-1.5 py-0'
   358→                        }
   359→                      >
   360→                        {TYPE_LABELS[provider.type]}
   361→                      </Badge>
   362→                    </div>
   363→                    <div className="flex items-center gap-3 text-xs text-gray-400">
   364→                      <span className="flex items-center gap-1 truncate">
   365→                        <ExternalLink className="w-3 h-3 shrink-0" />
   366→                        {provider.baseUrl}
   367→                      </span>
   368→                      <span className="flex items-center gap-1">
   369→                        <Shield className="w-3 h-3 shrink-0" />
   370→                        {AUTH_METHOD_LABELS[provider.authMethod]}
   371→                      </span>
   372→                      {provider.apiKey && (
   373→                        <span className="flex items-center gap-1">
   374→                          <Key className="w-3 h-3 shrink-0" />
   375→                          Configured
   376→                        </span>
   377→                      )}
   378→                    </div>
   379→                  </div>
   380→                  <div className="flex items-center gap-3 ml-4 shrink-0">
   381→                    <div className="flex items-center gap-1.5">
   382→                      <span className="text-[10px] text-gray-400">
   383→                        {provider.isActive ? 'Active' : 'Inactive'}
   384→                      </span>
   385→                      <Switch
   386→                        checked={provider.isActive}
   387→                        onCheckedChange={() => handleToggleActive(provider.id)}
   388→                        className="data-[state=checked]:bg-blue-600"
   389→                      />
   390→                    </div>
   391→                    <Separator orientation="vertical" className="h-6" />
   392→                    <Button
   393→                      variant="ghost"
   394→                      size="sm"
   395→                      className="h-7 w-7 p-0 text-gray-400 hover:text-red-500 hover:bg-red-50"
   396→                      onClick={() => handleDeleteProvider(provider.id)}
   397→                    >
   398→                      <Trash2 className="w-3.5 h-3.5" />
   399→                    </Button>
   400→                  </div>
   401→                </div>
   402→              ))}
   403→            </div>
   404→          )}
   405→        </CardContent>
   406→      </Card>
   407→
   408→      {/* ── Provider Add/Edit Dialog ── */}
   409→      <Dialog open={dialogOpen} onOpenChange={open => {
   410→        setDialogOpen(open)
   411→        if (!open) resetForm()
   412→      }}>
   413→        <DialogContent className="sm:max-w-lg">
   414→          <DialogHeader>
   415→            <DialogTitle>{editingProvider ? 'Edit Provider' : 'Add Lead Provider'}</DialogTitle>
   416→            <DialogDescription>
   417→              {editingProvider
   418→                ? 'Update the configuration for this lead data provider.'
   419→                : 'Configure a new lead data source for discovery and enrichment.'}
   420→            </DialogDescription>
   421→          </DialogHeader>
   422→
   423→          <div className="space-y-4 py-2">
   424→            {/* Provider Name */}
   425→            <div className="space-y-1.5">
   426→              <Label className="text-xs">Provider Name</Label>
   427→              <Input
   428→                placeholder="e.g. Serper (Google Search)"
   429→                value={formName}
   430→                onChange={e => setFormName(e.target.value)}
   431→              />
   432→            </div>
   433→
   434→            {/* Type */}
   435→            <div className="space-y-1.5">
   436→              <Label className="text-xs">Type</Label>
   437→              <Select value={formType} onValueChange={v => {
   438→                setFormType(v as LeadProvider['type'])
   439→                if (v === 'SERPER') {
   440→                  setFormBaseUrl('https://google.serper.dev')
   441→                  setFormAuthMethod('api-key')
   442→                  setFormAuthHeader('X-API-KEY')
   443→                }
   444→              }}>
   445→                <SelectTrigger className="w-full">
   446→                  <SelectValue placeholder="Select type" />
   447→                </SelectTrigger>
   448→                <SelectContent>
   449→                  <SelectItem value="SERPER">Serper</SelectItem>
   450→                  <SelectItem value="CUSTOM">Custom</SelectItem>
   451→                </SelectContent>
   452→              </Select>
   453→            </div>
   454→
   455→            {/* Base URL */}
   456→            <div className="space-y-1.5">
   457→              <Label className="text-xs">Base URL</Label>
   458→              <Input
   459→                placeholder="https://api.example.com"
   460→                value={formBaseUrl}
   461→                onChange={e => setFormBaseUrl(e.target.value)}
   462→              />
   463→            </div>
   464→
   465→            {/* Auth Method */}
   466→            <div className="space-y-1.5">
   467→              <Label className="text-xs">Auth Method</Label>
   468→              <Select value={formAuthMethod} onValueChange={v => setFormAuthMethod(v as LeadProvider['authMethod'])}>
   469→                <SelectTrigger className="w-full">
   470→                  <SelectValue placeholder="Select auth method" />
   471→                </SelectTrigger>
   472→                <SelectContent>
   473→                  <SelectItem value="bearer">Bearer Token</SelectItem>
   474→                  <SelectItem value="api-key">API Key in Header</SelectItem>
   475→                  <SelectItem value="header">Custom Header</SelectItem>
   476→                  <SelectItem value="none">None</SelectItem>
   477→                </SelectContent>
   478→              </Select>
   479→            </div>
   480→
   481→            {/* API Key (hidden when auth is none) */}
   482→            {formAuthMethod !== 'none' && (
   483→              <div className="space-y-1.5">
   484→                <Label className="text-xs">API Key</Label>
   485→                <Input
   486→                  type="password"
   487→                  placeholder={formAuthMethod === 'bearer' ? 'Bearer token...' : 'Enter your API key...'}
   488→                  value={formApiKey}
   489→                  onChange={e => setFormApiKey(e.target.value)}
   490→                />
   491→              </div>
   492→            )}
   493→
   494→            {/* Auth Header (only for api-key type) */}
   495→            {formAuthMethod === 'api-key' && (
   496→              <div className="space-y-1.5">
   497→                <Label className="text-xs">Auth Header Name</Label>
   498→                <Input
   499→                  placeholder="X-API-KEY"
   500→                  value={formAuthHeader}
   501→                  onChange={e => setFormAuthHeader(e.target.value)}
   502→                />
   503→              </div>
   504→            )}
   505→
   506→            {/* Custom Headers */}
   507→            <div className="space-y-1.5">
   508→              <Label className="text-xs">Custom Headers <span className="text-gray-400">(JSON)</span></Label>
   509→              <Textarea
   510→                placeholder={'{\n  "X-Custom-Header": "value"\n}'}
   511→                value={formHeaders}
   512→                onChange={e => setFormHeaders(e.target.value)}
   513→                className="min-h-[72px] font-mono text-xs"
   514→              />
   515→            </div>
   516→
   517→            {/* Custom Request Params */}
   518→            <div className="space-y-1.5">
   519→              <Label className="text-xs">Custom Request Params <span className="text-gray-400">(JSON)</span></Label>
   520→              <Textarea
   521→                placeholder={'{\n  "gl": "us",\n  "hl": "en"\n}'}
   522→                value={formParams}
   523→                onChange={e => setFormParams(e.target.value)}
   524→                className="min-h-[72px] font-mono text-xs"
   525→              />
   526→            </div>
   527→
   528→            {/* Active toggle */}
   529→            <div className="flex items-center justify-between py-1">
   530→              <Label className="text-xs">Active</Label>
   531→              <Switch
   532→                checked={formActive}
   533→                onCheckedChange={setFormActive}
   534→                className="data-[state=checked]:bg-blue-600"
   535→              />
   536→            </div>
   537→          </div>
   538→
   539→          <DialogFooter>
   540→            <Button
   541→              variant="outline"
   542→              size="sm"
   543→              onClick={() => {
   544→                setDialogOpen(false)
   545→                resetForm()
   546→              }}
   547→            >
   548→              Cancel
   549→            </Button>
   550→            <Button
   551→              size="sm"
   552→              className="bg-blue-600 hover:bg-blue-700 text-white"
   553→              disabled={savingProviders || !formName.trim() || !formBaseUrl.trim()}
   554→              onClick={handleSaveProvider}
   555→            >
   556→              {savingProviders ? (
   557→                <>
   558→                  <Settings className="w-3.5 h-3.5 mr-1 animate-spin" />
   559→                  Saving...
   560→                </>
   561→              ) : (
   562→                <>
   563→                  <CheckCircle className="w-3.5 h-3.5 mr-1" />
   564→                  Save Provider
   565→                </>
   566→              )}
   567→            </Button>
   568→          </DialogFooter>
   569→        </DialogContent>
   570→      </Dialog>
   571→
   572→      {/* Profile */}
   573→      <Card className="border-slate-200">
   574→        <CardHeader className="pb-3"><CardTitle className="text-sm font-semibold">Profile</CardTitle></CardHeader>
   575→        <CardContent className="space-y-3">
   576→          <div><Label>Display Name</Label><Input className="mt-1" value={name} onChange={e => setName(e.target.value)} /></div>
   577→          <Button size="sm" onClick={updateProfile} disabled={loading || !name}>
   578→            {saved ? <><CheckCircle className="w-3.5 h-3.5 mr-1" />Saved</> : <><Save className="w-3.5 h-3.5 mr-1" />Update Profile</>}
   579→          </Button>
   580→        </CardContent>
   581→      </Card>
   582→
   583→      {/* Password */}
